// Dashboard par PKR balance update karne ka function function syncBalance() { // LocalStorage se balance uthana, agar nahi hai to 0.00 let currentBalance = localStorage.getItem('userPKR') || "0.00"; // UI update karna document.getElementById('dashboard-coins').innerText = `Rs. ${parseFloat(currentBalance).toFixed(2)}`; document.getElementById('coin-balance-display').innerHTML = ` PKR: ${parseFloat(currentBalance).toFixed(2)} `; } // Page load hote hi sync karein window.onload = syncBalance; // Agar user dusri tab mein game khel raha ho, to balance auto-update hoga window.addEventListener('storage', (e) => { if (e.key === 'userPKR') { syncBalance(); } });
